# Function to get the BigQuery configuration from XCom
def run_bigquery_insert_job_{{id}}(**kwargs):
    ti = kwargs['ti']
    config = ti.xcom_pull(task_ids='prepare_config_{{id}}', key='bigquery_config')
    insert_job = BigQueryInsertJobOperator(
        task_id='run_query_task_{{id}}',
        configuration=config,
        location='{{bqRegion}}',  # Specify your BigQuery location
        impersonate_service_account=service_account,
        dag=kwargs['dag'],
    )
    insert_job.execute(context=kwargs)

# Task to run the BigQuery job
run_query_task_{{id}} = PythonOperator(
    task_id='run_query_task_{{id}}',
    python_callable=run_bigquery_insert_job_{{id}},
    provide_context=True,
    dag=dag,
)
